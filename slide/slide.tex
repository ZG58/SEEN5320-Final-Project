\begin{frame}{Why VPSA Needs Surrogate Optimization}
\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.48\textwidth}
\vspace{0pt}
\begin{block}{Process challenge}
\small
\begin{itemize}
    \item Dry flue gas is dilute: 15/85 mol\% CO$_2$/N$_2$ in the reference case.
    \item Purity, recovery, productivity, and energy move in competing directions.
    \item A cyclic VPSA model is expensive to optimize directly.
\end{itemize}
\end{block}
\source{Wang et al. (2022); Ruthven (1984)}
\end{column}
\begin{column}{0.48\textwidth}
\vspace{0pt}
\begin{block}{This project}
\small
\begin{itemize}
    \item Use successful detailed simulations as training data.
    \item Benchmark several surrogate families with the same split.
    \item Search Pareto fronts after validating the selected model.
\end{itemize}
\end{block}
\source{Deb et al. (2002); Wang et al. (2022)}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Reference VPSA Cycle}

\vspace{0pt}

\begin{block}{Reference case}
\begin{itemize}

    \item Configuration: 3 beds, 7 cyclic steps.
    \item Feed: dry flue gas with 15 mol\% CO$_2$.
    \item Sampling: Latin hypercube design around six operating variables.

\end{block}

\begin{frame}{Decision Variables}
\scriptsize
\begin{tabularx}{\textwidth}{lY Y}
\toprule
Variable & Cycle location & Main effect \\
\midrule
FAD & Adsorption feed & Bed loading rate and N$_2$ breakthrough tendency \\
FRP & Replacement flow & CO$_2$-rich purge strength for removing residual N$_2$ \\
FVU & Vacuum flow & Desorption rate and vacuum-pump energy demand \\
TRP & Replacement time & Purity improvement versus cycle-time loss \\
TED & Equalization depressurization & Gas recovery and pressure-energy reuse \\
TPR & Pressurization & Initial pressure state before the next adsorption step \\
\bottomrule
\end{tabularx}
\takeaway{The six variables are coupled because each bed enters the next step with the pressure, loading, and composition left by the previous step.}

\begin{frame}{Conclusions}
\small
\begin{itemize}
    \item The raw MAT file was converted into a reproducible scalar dataset with 1708 successful VPSA simulations.
    \item Eight surrogate families were compared using global and Pareto-focused validation.
    \item RBF kernel ridge gave the best optimization-oriented score and was used inside NSGA-II.
    \item The resulting Pareto fronts expose the main purity-recovery and productivity-energy tradeoffs.
\end{itemize}
\takeaway{The workflow is ready for final confirmation by rerunning selected nondominated candidates in the detailed simulator.}
\end{frame}
